
USER_CREATION
--------------

select 
'GRANT   SELECT,INSERT,UPDATE ON '+ TABLE_SCHEMA+'.'+ TABLE_NAME+' TO DEV_USER;' 
from INFORMATION_SCHEMA.TABLES
where
TABLE_SCHEMA IN ('DBO','CDB','FEP','MDMS','NMS','CP','DWH','KETTLE','DSM','DG','PLM','STGMDM') 
AND TABLE_NAME NOT LIKE '%2018%'


SELECT DISTINCT  TABLE_SCHEMA FROM INFORMATION_SCHEMA.TABLES
